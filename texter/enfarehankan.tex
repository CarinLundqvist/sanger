\begin{song}{En F-are han kan}{enfarehankan}
\mel{Tre trallande jäntor}
\begin{vers}
En F-are han kan\\
ta supen som en man.\\
Den supen i strupen på glupen\\
den försvann.\\
Och mera vill han ha\\
kan ej för mycket ta.\\
En F-are blir aldrig full,\\
nej han blir bare gla'! \\
\end{vers}
\end{song}
